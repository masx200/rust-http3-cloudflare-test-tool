18fn js_fetch(req: &web_sys::Request) -> Promise {
19    use wasm_bindgen::{JsCast, JsValue};
20    let global = js_sys::global();
21
22    if let Ok(true) = js_sys::Reflect::has(&global, &JsValue::from_str("ServiceWorkerGlobalScope"))
23    {
24        global
25            .unchecked_into::<web_sys::ServiceWorkerGlobalScope>()
26            .fetch_with_request(req)
27    } else {
28        // browser
29        fetch_with_request(req)
30    }
31}

141    // merge request headers with Client default_headers, prior to external http fetch
142    fn merge_headers(&self, req: &mut Request) {
143        use http::header::Entry;
144        let headers: &mut HeaderMap = req.headers_mut();
145        // insert default headers in the request headers
146        // without overwriting already appended headers.
147        for (key, value) in self.config.headers.iter() {
148            if let Entry::Vacant(entry) = headers.entry(key) {
149                entry.insert(value.clone());
150            }
151        }
152    }
153
154    pub(super) fn execute_request(
155        &self,
156        mut req: Request,
157    ) -> impl Future<Output = crate::Result<Response>> {
158        self.merge_headers(&mut req);
159        fetch(req)
160    }

185// Can use new methods in web-sys when requiring v0.2.93.
186// > `init.method(m)` to `init.set_method(m)`
187// For now, ignore their deprecation.
188#[allow(deprecated)]
189async fn fetch(req: Request) -> crate::Result<Response> {
190    // Build the js Request
191    let mut init = web_sys::RequestInit::new();
192    init.method(req.method().as_str());
193
194    // convert HeaderMap to Headers
195    let js_headers = web_sys::Headers::new()
196        .map_err(crate::error::wasm)
197        .map_err(crate::error::builder)?;
198
199    for (name, value) in req.headers() {
200        js_headers
201            .append(
202                name.as_str(),
203                value.to_str().map_err(crate::error::builder)?,
204            )
205            .map_err(crate::error::wasm)
206            .map_err(crate::error::builder)?;
207    }
208    init.headers(&js_headers.into());
209
210    // When req.cors is true, do nothing because the default mode is 'cors'
211    if !req.cors {
212        init.mode(web_sys::RequestMode::NoCors);
213    }
214
215    if let Some(creds) = req.credentials {
216        init.credentials(creds);
217    }
218
219    if let Some(cache) = req.cache {
220        init.set_cache(cache);
221    }
222
223    if let Some(body) = req.body() {
224        if !body.is_empty() {
225            init.body(Some(body.to_js_value()?.as_ref()));
226        }
227    }
228
229    let mut abort = AbortGuard::new()?;
230    if let Some(timeout) = req.timeout() {
231        abort.timeout(*timeout);
232    }
233    init.signal(Some(&abort.signal()));
234
235    let js_req = web_sys::Request::new_with_str_and_init(req.url().as_str(), &init)
236        .map_err(crate::error::wasm)
237        .map_err(crate::error::builder)?;
238
239    // Await the fetch() promise
240    let p = js_fetch(&js_req);
241    let js_resp = super::promise::<web_sys::Response>(p)
242        .await
243        .map_err(|error| {
244            if error.to_string() == "JsValue(\"reqwest::errors::TimedOut\")" {
245                crate::error::TimedOut.into()
246            } else {
247                error
248            }
249        })
250        .map_err(crate::error::request)?;
251
252    // Convert from the js Response
253    let mut resp = http::Response::builder().status(js_resp.status());
254
255    let url = Url::parse(&js_resp.url()).expect_throw("url parse");
256
257    let js_headers = js_resp.headers();
258    let js_iter = js_sys::try_iter(&js_headers)
259        .expect_throw("headers try_iter")
260        .expect_throw("headers have an iterator");
261
262    for item in js_iter {
263        let item = item.expect_throw("headers iterator doesn't throw");
264        let serialized_headers: String = JSON::stringify(&item)
265            .expect_throw("serialized headers")
266            .into();
267        let [name, value]: [String; 2] = serde_json::from_str(&serialized_headers)
268            .expect_throw("deserializable serialized headers");
269        resp = resp.header(&name, &value);
270    }
271
272    resp.body(js_resp)
273        .map(|resp| Response::new(resp, url, abort))
274        .map_err(crate::error::request)
275}

287    /// Returns a 'Client' that uses this ClientBuilder configuration
288    pub fn build(mut self) -> Result<Client, crate::Error> {
289        if let Some(err) = self.config.error {
290            return Err(err);
291        }
292
293        let config = std::mem::take(&mut self.config);
294        Ok(Client {
295            config: Arc::new(config),
296        })
297    }
298
299    /// Sets the `User-Agent` header to be used by this client.
300    pub fn user_agent<V>(mut self, value: V) -> ClientBuilder
301    where
302        V: TryInto<HeaderValue>,
303        V::Error: Into<http::Error>,
304    {
305        match value.try_into() {
306            Ok(value) => {
307                self.config.headers.insert(USER_AGENT, value);
308            }
309            Err(e) => {
310                self.config.error = Some(crate::error::builder(e.into()));
311            }
312        }
313        self
314    }
315
316    /// Sets the default headers for every request
317    pub fn default_headers(mut self, headers: HeaderMap) -> ClientBuilder {
318        for (key, value) in headers.iter() {
319            self.config.headers.insert(key, value.clone());
320        }
321        self
322    }
323}

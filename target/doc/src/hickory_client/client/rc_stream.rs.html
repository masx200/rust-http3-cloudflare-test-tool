8use std::pin::Pin;
9use std::sync::Arc;
10use std::task::{Context, Poll};
11
12use futures_util::lock::Mutex;
13use futures_util::stream::{Fuse, Stream, StreamExt};
14use futures_util::{FutureExt, ready};
15
16#[allow(clippy::type_complexity)]
17#[must_use = "stream do nothing unless polled"]
18pub(crate) struct RcStream<S: Stream>
19where
20    S: Stream + Send + Unpin,
21    S::Item: Clone + Send,
22{
23    stream_and_result: Arc<Mutex<(Fuse<S>, Vec<S::Item>)>>,
24    pos: usize,
25}
26
27pub(crate) fn rc_stream<S>(stream: S) -> RcStream<S>
28where
29    S: Stream + Unpin + Send,
30    S::Item: Clone + Send,
31{
32    let stream_and_result = Arc::new(Mutex::new((stream.fuse(), vec![])));
33
34    RcStream {
35        stream_and_result,
36        pos: 0,
37    }
38}
39
40impl<S> Stream for RcStream<S>
41where
42    S: Stream + Send + Unpin,
43    S::Item: Clone + Send,
44{
45    type Item = S::Item;
46
47    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
48        // try and get a mutable reference to execute the future
49        // at least one caller should be able to get a mut reference... others will
50        //  wait for it to complete.
51        let mut stream_and_result = ready!(self.stream_and_result.lock().poll_unpin(cx));
52        let (stream, stored_result) = &mut *stream_and_result;
53        if stored_result.len() > self.pos {
54            let result = stored_result[self.pos].clone();
55            drop(stream_and_result);
56            self.pos += 1;
57            return Poll::Ready(Some(result));
58        }
59
60        // if pending it's either done, or it's actually pending
61        match stream.poll_next_unpin(cx) {
62            Poll::Pending => Poll::Pending,
63            Poll::Ready(result) => {
64                if let Some(result) = &result {
65                    stored_result.push(result.clone());
66                }
67                Poll::Ready(result)
68            }
69        }
70    }
71}
72
73impl<S> Clone for RcStream<S>
74where
75    S: Stream + Send + Unpin,
76    S::Item: Clone + Send + Unpin,
77{
78    fn clone(&self) -> Self {
79        Self {
80            stream_and_result: Arc::clone(&self.stream_and_result),
81            pos: 0, // index is not kept to allow to read first messages
82        }
83    }
84}

86#[cfg(test)]
87mod tests {
88    use futures::executor::block_on;
89    use futures::future;
90    use futures_util::stream::once;
91
92    use crate::proto::xfer::FirstAnswer;
93    use crate::proto::{ProtoError, ProtoErrorKind};
94
95    use super::*;
96
97    #[test]
98    fn test_rc_stream() {
99        let future = future::ok::<usize, ProtoError>(1_usize);
100
101        let rc = rc_stream(once(future));
102
103        let i = block_on(rc.clone().first_answer()).unwrap();
104        assert_eq!(i, 1);
105
106        let i = block_on(rc.first_answer()).unwrap();
107        assert_eq!(i, 1);
108    }
109
110    #[test]
111    fn test_rc_stream_failed() {
112        let future = future::err::<usize, ProtoError>(ProtoError::from(ProtoErrorKind::Busy));
113
114        let rc = rc_stream(once(future));
115
116        let i = block_on(rc.clone().first_answer()).unwrap_err();
117        assert!(matches!(i.kind(), ProtoErrorKind::Busy));
118
119        let i = block_on(rc.first_answer()).unwrap_err();
120        assert!(matches!(i.kind(), ProtoErrorKind::Busy));
121    }
122}

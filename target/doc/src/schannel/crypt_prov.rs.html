13/// A CryptoAPI handle to a provider of a key.
14pub struct CryptProv(Cryptography::HCRYPTPROV_OR_NCRYPT_KEY_HANDLE);
15
16impl Drop for CryptProv {
17    fn drop(&mut self) {
18        unsafe {
19            Cryptography::CryptReleaseContext(self.0, 0);
20        }
21    }
22}
23
24inner!(CryptProv, Cryptography::HCRYPTPROV_OR_NCRYPT_KEY_HANDLE);
25
26impl CryptProv {
27    /// Imports a key into this provider.
28    pub fn import(&mut self) -> ImportOptions {
29        ImportOptions {
30            prov: self,
31            flags: 0,
32        }
33    }
34}

36/// A builder for `CryptProv`s.
37#[derive(Default)]
38pub struct AcquireOptions {
39    container: Option<Vec<u16>>,
40    provider: Option<Vec<u16>>,
41    flags: u32,
42}
43
44impl AcquireOptions {
45    /// Returns a new builder with default settings.
46    pub fn new() -> AcquireOptions {
47        AcquireOptions::default()
48    }
49
50    /// Sets the name for this key container.
51    ///
52    /// This should not be set if `verify_context` is set.
53    pub fn container(&mut self, container: &str) -> &mut AcquireOptions {
54        self.container = Some(container.encode_utf16().chain(Some(0)).collect());
55        self
56    }
57
58    /// Sets the name of the CSP to be used.
59    pub fn provider(&mut self, provider: &str) -> &mut AcquireOptions {
60        self.provider = Some(provider.encode_utf16().chain(Some(0)).collect());
61        self
62    }
63
64    /// If set, private keys will not be accessible or persisted.
65    pub fn verify_context(&mut self, verify_context: bool) -> &mut AcquireOptions {
66        self.flag(Cryptography::CRYPT_VERIFYCONTEXT, verify_context)
67    }
68
69    /// If set, the container will be created.
70    pub fn new_keyset(&mut self, new_keyset: bool) -> &mut AcquireOptions {
71        self.flag(Cryptography::CRYPT_NEWKEYSET, new_keyset)
72    }
73
74    /// If set, the container will be stored as a machine rather than user keys.
75    pub fn machine_keyset(&mut self, machine_keyset: bool) -> &mut AcquireOptions {
76        self.flag(Cryptography::CRYPT_MACHINE_KEYSET, machine_keyset)
77    }
78
79    /// If set, an error will be returned if user intervention is required
80    /// rather than displaying a dialog.
81    pub fn silent(&mut self, silent: bool) -> &mut AcquireOptions {
82        self.flag(Cryptography::CRYPT_SILENT, silent)
83    }
84
85    fn flag(&mut self, flag: u32, on: bool) -> &mut AcquireOptions {
86        if on {
87            self.flags |= flag;
88        } else {
89            self.flags &= !flag;
90        }
91
92        self
93    }
94
95    /// Acquires a container.
96    pub fn acquire(&self, type_: ProviderType) -> io::Result<CryptProv> {
97        unsafe {
98            let container = self
99                .container
100                .as_ref()
101                .map(|s| s.as_ptr())
102                .unwrap_or(ptr::null());
103            let provider = self
104                .provider
105                .as_ref()
106                .map(|s| s.as_ptr())
107                .unwrap_or(ptr::null());
108
109            let mut prov = 0;
110            let res = Cryptography::CryptAcquireContextW(
111                &mut prov,
112                container as *mut _,
113                provider as *mut _,
114                type_.0,
115                self.flags,
116            );
117            if res != 0 {
118                Ok(CryptProv(prov))
119            } else {
120                Err(io::Error::last_os_error())
121            }
122        }
123    }
124}

178/// A builder for key imports.
179pub struct ImportOptions<'a> {
180    prov: &'a mut CryptProv,
181    flags: u32,
182}
183
184impl<'a> ImportOptions<'a> {
185    /// Imports a DER-encoded PKCS1 private key.
186    pub fn import(&mut self, der: &[u8]) -> io::Result<CryptKey> {
187        unsafe {
188            assert!(der.len() <= u32::max_value() as usize);
189            let mut buf = ptr::null_mut();
190            let mut len = 0;
191            let res = Cryptography::CryptDecodeObjectEx(
192                Cryptography::X509_ASN_ENCODING | Cryptography::PKCS_7_ASN_ENCODING,
193                Cryptography::PKCS_RSA_PRIVATE_KEY,
194                der.as_ptr(),
195                der.len() as u32,
196                Cryptography::CRYPT_DECODE_ALLOC_FLAG,
197                ptr::null_mut(),
198                &mut buf as *mut _ as *mut c_void,
199                &mut len,
200            );
201            if res == 0 {
202                return Err(io::Error::last_os_error());
203            }
204
205            let mut key = 0;
206            let res = Cryptography::CryptImportKey(self.prov.0, buf, len, 0, self.flags, &mut key);
207            Foundation::LocalFree(buf as *mut _);
208
209            if res != 0 {
210                Ok(CryptKey::from_inner(key))
211            } else {
212                Err(io::Error::last_os_error())
213            }
214        }
215    }
216
217    /// Imports a DER-encoded PKCS8 private key.
218    pub fn import_pkcs8(&mut self, der: &[u8]) -> io::Result<CryptKey> {
219        unsafe {
220            assert!(der.len() <= u32::max_value() as usize);
221
222            // Decode the der format into a CRYPT_PRIVATE_KEY_INFO struct
223            let mut buf = ptr::null_mut();
224            let mut len = 0;
225            let res = Cryptography::CryptDecodeObjectEx(
226                Cryptography::X509_ASN_ENCODING | Cryptography::PKCS_7_ASN_ENCODING,
227                Cryptography::PKCS_PRIVATE_KEY_INFO,
228                der.as_ptr(),
229                der.len() as u32,
230                Cryptography::CRYPT_DECODE_ALLOC_FLAG,
231                ptr::null_mut(),
232                &mut buf as *mut _ as *mut c_void,
233                &mut len,
234            );
235            if res == 0 {
236                return Err(io::Error::last_os_error());
237            }
238            let pkey: Cryptography::CRYPT_PRIVATE_KEY_INFO = *buf;
239            let pkey = pkey.PrivateKey;
240
241            let res = self.import(slice::from_raw_parts(pkey.pbData, pkey.cbData as usize));
242            Foundation::LocalFree(buf as *mut _);
243            res
244        }
245    }
246
247    /// Imports a PEM-encoded PKCS8 private key.
248    /// This functions decodes PEM blocks with or without "-----BEGIN PRIVATE KEY-----"
249    /// and "-----END PRIVATE KEY-----" headers, but if PEM guards are present they must be exactly
250    /// these.
251    pub fn import_pkcs8_pem(&mut self, pem: &[u8]) -> io::Result<CryptKey> {
252        let pem_str = std::str::from_utf8(pem)
253            .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid utf-8"))?
254            .trim();
255
256        if pem_str.starts_with("-----")
257            && (!pem_str.starts_with("-----BEGIN PRIVATE KEY-----")
258                || !pem_str.ends_with("-----END PRIVATE KEY-----"))
259        {
260            return Err(io::Error::new(
261                io::ErrorKind::InvalidData,
262                "expected '-----BEGIN PRIVATE KEY-----'\
263                                          and '-----END PRIVATE KEY-----' PEM guards",
264            ));
265        }
266        unsafe {
267            assert!(pem.len() <= u32::max_value() as usize);
268
269            // Decode the pem wrapper before passing it to import_pkcs8
270            // Call once first to figure out the necessary buffer size
271            let mut len = 0;
272            let res = Cryptography::CryptStringToBinaryA(
273                pem.as_ptr(),
274                pem.len() as u32,
275                Cryptography::CRYPT_STRING_BASE64_ANY,
276                ptr::null_mut(),
277                &mut len,
278                ptr::null_mut(),
279                ptr::null_mut(),
280            );
281            if res == 0 {
282                return Err(io::Error::last_os_error());
283            }
284
285            // Call second time to actually get the DER bytes
286            let mut der_buf = vec![0; len as usize];
287            let res = Cryptography::CryptStringToBinaryA(
288                pem.as_ptr(),
289                pem.len() as u32,
290                Cryptography::CRYPT_STRING_BASE64_ANY,
291                der_buf.as_mut_ptr(),
292                &mut len,
293                ptr::null_mut(),
294                ptr::null_mut(),
295            );
296            if res == 0 {
297                return Err(io::Error::last_os_error());
298            }
299            self.import_pkcs8(&der_buf)
300        }
301    }
302}

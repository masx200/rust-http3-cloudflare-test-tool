1use std::alloc;
2use std::mem;
3use std::ptr;
4use std::slice;
5
6use windows_sys::Win32::Security::Authentication::Identity;
7
8// This is manually calculated here rather than using `size_of::<SEC_APPLICATION_PROTOCOL_LIST>()`,
9// as the latter is 2 bytes too large because it accounts for padding at the end of the struct for
10// alignment requirements, which is irrelevant in actual usage because there is a variable-length
11// array at the end of the struct.
12const SEC_APPLICATION_PROTOCOL_LIST_HEADER_SIZE: usize =
13    mem::size_of::<u32>() + mem::size_of::<u16>();
14const SEC_APPLICATION_PROTOCOL_HEADER_SIZE: usize =
15    mem::size_of::<u32>() + SEC_APPLICATION_PROTOCOL_LIST_HEADER_SIZE;
16
17pub struct AlpnList {
18    layout: alloc::Layout,
19    memory: ptr::NonNull<u8>,
20}
21
22impl Drop for AlpnList {
23    fn drop(&mut self) {
24        unsafe {
25            // Safety: `self.memory` was allocated with `self.layout` and is non-null.
26            alloc::dealloc(self.memory.as_ptr(), self.layout);
27        }
28    }
29}
30
31impl AlpnList {
32    pub fn new(protos: &[Vec<u8>]) -> Self {
33        // ALPN wire format is each ALPN preceded by its length as a byte.
34        let mut alpn_wire_format =
35            Vec::with_capacity(protos.iter().map(Vec::len).sum::<usize>() + protos.len());
36        for alpn in protos {
37            alpn_wire_format.push(alpn.len() as u8);
38            alpn_wire_format.extend(alpn);
39        }
40
41        let size = SEC_APPLICATION_PROTOCOL_HEADER_SIZE + alpn_wire_format.len();
42        let layout = alloc::Layout::from_size_align(
43            size,
44            mem::align_of::<Identity::SEC_APPLICATION_PROTOCOLS>(),
45        )
46        .unwrap();
47
48        unsafe {
49            // Safety: `layout` is guaranteed to have non-zero size.
50            let memory = match ptr::NonNull::new(alloc::alloc(layout)) {
51                Some(ptr) => ptr,
52                None => alloc::handle_alloc_error(layout),
53            };
54
55            // Safety: `memory` was created from `layout`.
56            let buf = slice::from_raw_parts_mut(memory.as_ptr(), layout.size());
57            let protocols = &mut *(buf.as_mut_ptr() as *mut Identity::SEC_APPLICATION_PROTOCOLS);
58            protocols.ProtocolListsSize =
59                (SEC_APPLICATION_PROTOCOL_LIST_HEADER_SIZE + alpn_wire_format.len()) as u32;
60
61            let protocol = &mut *protocols.ProtocolLists.as_mut_ptr();
62            protocol.ProtoNegoExt = Identity::SecApplicationProtocolNegotiationExt_ALPN;
63            protocol.ProtocolListSize = alpn_wire_format.len() as u16;
64
65            let protocol_list_offset =
66                protocol.ProtocolList.as_ptr() as usize - buf.as_ptr() as usize;
67            let protocol_list = &mut buf[protocol_list_offset..];
68            protocol_list.copy_from_slice(&alpn_wire_format);
69
70            Self { layout, memory }
71        }
72    }
73}

75impl std::ops::Deref for AlpnList {
76    type Target = [u8];
77
78    fn deref(&self) -> &Self::Target {
79        unsafe {
80            // Safety: `self.memory` was created from `self.layout`.
81            slice::from_raw_parts(self.memory.as_ptr(), self.layout.size())
82        }
83    }
84}
85

88        unsafe {
89            // Safety: `self.memory` was created from `self.layout`.

91        }
92    }

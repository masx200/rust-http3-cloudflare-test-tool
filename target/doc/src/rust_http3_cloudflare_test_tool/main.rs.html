69// --- Helper: 提取 A/AAAA 记录的 IP ---
70// 提取 A/AAAA 记录 IP 的公共逻辑，可用于 answers, authorities, additionals 三个部分。
71fn extract_a_aaaa_ips(records: &[trust_dns_resolver::proto::rr::Record], ips: &mut HashSet<IpAddr>) {
72    for record in records {
73        // rdata.ip_addr() 是 trust-dns 中用于 A/AAAA 记录的便捷方法
74        if let Some(ip) = record.data().and_then(|rdata| rdata.ip_addr()) {
75            ips.insert(ip);
76        }
77    }
78}

80// --- Helper: 手动 DoH 查询函数 (使用 reqwest + trust-dns proto) ---
81async fn doh_query_manual(
82    client: &Client,
83    doh_url: &str,
84    domain: &str,
85    record_type: RecordType,
86) -> Result<Message> {
87    // 1. 构建 DNS 查询消息 (Question)
88    let mut message = Message::new();
89    let name = Name::from_str(domain).context("Invalid domain name for DNS query")?;
90
91    let query = Query::query(name, record_type);
92    message.add_query(query);
93
94    // 2. 消息编码为二进制 - 使用 trust-dns-resolver 的高级 API
95    let request_buffer = message.to_vec().context("Failed to encode DNS message")?;
96
97    // 3. 使用 reqwest 发送 POST 请求
98    let response = client
99        .post(doh_url)
100        .header("Content-Type", "application/dns-message")
101        .header("Accept", "application/dns-message")
102        .body(request_buffer)
103        .send()
104        .await
105        .context(format!("Failed to send DoH request to {}", doh_url))?;
106
107    if !response.status().is_success() {
108        // 如果 HTTP 状态码不是 2xx，返回错误
109        return Err(anyhow::anyhow!(
110            "DoH server returned HTTP error status: {}",
111            response.status()
112        ));
113    }
114
115    // 4. 读取二进制响应体
116    let response_body = response
117        .bytes()
118        .await
119        .context("Failed to read DoH response body")?;
120
121    // 5. 解码 DNS 响应消息
122    // 修复：使用 Message::from_vec 方法直接从字节数组解析
123    let dns_response = Message::from_vec(&response_body)
124        .context("Failed to decode DNS binary message (invalid data)")?;
125
126    Ok(dns_response)
127}

129// --- 4. 核心：DoH HTTPS 记录查询 (RFC 8484 Binary) ---
130async fn resolve_https_record(client: &Client, doh_url: &str, domain: &str) -> Result<Vec<IpAddr>> {
131    let mut ips = HashSet::new();
132
133    // 1. 查询 HTTPS (SVCB) 记录
134    match doh_query_manual(client, doh_url, domain, RecordType::HTTPS).await {
135        Ok(response) => {
136            for record in response.answers() {
137                if let Some(rdata) = record.data() {
138                    // 检查是否是 SVCB 记录并提取 IP hints
139                    if let RData::SVCB(svc_rec) = rdata {
140                        // 从 ipv4hint 中提取 IP
141                        if let Some(ipv4_param) = svc_rec.svc_params().get(&SvcParamKey::Ipv4Hint) {
142                            if let Some(ip_list) = ipv4_param.ip_addrs() {
143                                for ip in ip_list {
144                                    ips.insert(*ip);
145                                }
146                            }
147                        }
148
149                        // 从 ipv6hint 中提取 IP
150                        if let Some(ipv6_param) = svc_rec.svc_params().get(&SvcParamKey::Ipv6Hint) {
151                            if let Some(ip_list) = ipv6_param.ip_addrs() {
152                                for ip in ip_list {
153                                    ips.insert(*ip);
154                                }
155                            }
156                        }
157                    }
158                }
159            }
160        },
161        Err(e) => eprintln!("    [X] HTTPS记录解析失败: {:?}", e),
162    }
163
164    // 2. 如果未在 SVCB 记录中找到 IP，执行 A/AAAA 记录的兜底查询
165    if ips.is_empty() {
166        println!("    -> HTTPS记录中未找到 IP，尝试 A/AAAA 记录查询作为兜底...");
167
168        // 查询 A 记录
169        if let Ok(response) = doh_query_manual(client, doh_url, domain, RecordType::A).await {
170            // 检查 Answers, Authorities, Additionals 所有部分
171            extract_a_aaaa_ips(response.answers(), &mut ips);
172            extract_a_aaaa_ips(response.authoritative(), &mut ips);
173            extract_a_aaaa_ips(response.additionals(), &mut ips);
174        }
175
176        // 查询 AAAA 记录
177        if let Ok(response) = doh_query_manual(client, doh_url, domain, RecordType::AAAA).await {
178            extract_a_aaaa_ips(response.answers(), &mut ips);
179            extract_a_aaaa_ips(response.authoritative(), &mut ips);
180            extract_a_aaaa_ips(response.additionals(), &mut ips);
181        }
182    }
183
184    // 3. 排序并返回
185    let mut ip_vec = ips.into_iter().collect::<Vec<_>>();
186    // 排序逻辑 (这里默认 IPv4 优先)
187    ip_vec.sort_by_key(|ip| ip.is_ipv6());
188
189    Ok(ip_vec)
190}

203// --- 7. HTTP/3 连通性测试 ---
204async fn test_connectivity(task: InputTask, ip: IpAddr, dns_source: String) -> TestResult {
205    // 关键点 1: URL 决定 SNI。reqwest 将使用 test_sni_host 作为 SNI。
206    let url = format!("https://{}:{}/", task.test_sni_host, task.port);
207
208    let socket_addr = SocketAddr::new(ip, task.port);
209    let ip_ver = if ip.is_ipv6() { "IPv6" } else { "IPv4" };
210
211    // 创建独立的 Client 实例以绑定特定的 IP
212    let client_build = Client::builder()
213        // 关键点 2: 强制 IP 连接 (将 test_sni_host 解析到特定的 IP)
214        .resolve_to_addrs(&task.test_sni_host, &[socket_addr])
215        .danger_accept_invalid_certs(true)
216        .http3_prior_knowledge() // 核心：强制使用 HTTP/3
217        .timeout(std::time::Duration::from_secs(5))
218        .no_proxy()
219        .build();
220
221    let client = match client_build {
222        Ok(c) => c,
223        Err(e) => {
224            return TestResult::fail(&task, &ip.to_string(), ip_ver, e.to_string(), dns_source)
225        }
226    };
227
228    let start = Instant::now();
229
230    // 发送请求
231    let req = client
232        .get(&url)
233        // 关键点 3: 覆盖 Host Header
234        .header("Host", &task.test_host_header)
235        .header("User-Agent", "curl/8.12.1")
236        .send();
237
238    match req.await {
239        Ok(res) => {
240            let latency = start.elapsed().as_millis() as u64;
241            let status = res.status().as_u16();
242            let server = res
243                .headers()
244                .get("server")
245                .and_then(|v| v.to_str().ok())
246                .map(|s| s.to_string());
247
248            let protocol = match res.version() {
249                Version::HTTP_3 => "h3",
250                Version::HTTP_2 => "h2",
251                Version::HTTP_11 => "http/1.1",
252                _ => "unknown",
253            };
254
255            TestResult {
256                domain_used: task.doh_resolve_domain,
257                target_ip: ip.to_string(),
258                ip_version: ip_ver.to_string(),
259                sni_host: task.test_sni_host,
260                host_header: task.test_host_header,
261                success: status < 500,
262                status_code: Some(status),
263                protocol: protocol.to_string(),
264                latency_ms: Some(latency),
265                server_header: server,
266                error_msg: None,
267                dns_source,
268            }
269        }
270        Err(e) => TestResult::fail(&task, &ip.to_string(), ip_ver, e.to_string(), dns_source),
271    }
272}

274impl TestResult {
275    fn fail(task: &InputTask, ip: &str, ver: &str, msg: String, dns_source: String) -> Self {
276        TestResult {
277            domain_used: task.doh_resolve_domain.clone(),
278            target_ip: ip.to_string(),
279            ip_version: ver.to_string(),
280            sni_host: task.test_sni_host.clone(),
281            host_header: task.test_host_header.clone(),
282            success: false,
283            status_code: None,
284            protocol: "none".to_string(),
285            latency_ms: None,
286            server_header: None,
287            error_msg: Some(msg),
288            dns_source,
289        }
290    }
291}

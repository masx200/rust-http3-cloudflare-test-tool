3pub struct OwnedPcwstr(Vec<u16>);
4
5pub fn pcwstr<T: AsRef<str>>(value: T) -> OwnedPcwstr {
6    OwnedPcwstr(
7        value
8            .as_ref()
9            .encode_utf16()
10            .chain(core::iter::once(0))
11            .collect(),
12    )
13}
14
15pub fn multi_pcwstr<T: AsRef<str>>(value: &[T]) -> OwnedPcwstr {
16    OwnedPcwstr(
17        value
18            .iter()
19            .flat_map(|value| value.as_ref().encode_utf16().chain(core::iter::once(0)))
20            .chain(core::iter::once(0))
21            .collect(),
22    )
23}
24
25impl OwnedPcwstr {
26    pub fn as_ptr(&self) -> *const u16 {
27        debug_assert!(
28            self.0.last() == Some(&0),
29            "`OwnedPcwstr` isn't null-terminated"
30        );
31        self.0.as_ptr()
32    }
33
34    // Get the string as 8-bit bytes including the two terminating null bytes.
35    pub fn as_bytes(&self) -> &[u8] {
36        unsafe { core::slice::from_raw_parts(self.as_ptr() as *const _, self.0.len() * 2) }
37    }
38
39    pub fn as_raw(&self) -> PCWSTR {
40        PCWSTR(self.as_ptr())
41    }

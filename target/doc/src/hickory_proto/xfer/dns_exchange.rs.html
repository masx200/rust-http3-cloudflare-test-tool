81/// This is a generic Exchange implemented over multiplexed DNS connection providers.
82///
83/// The underlying `DnsRequestSender` is expected to multiplex any I/O connections. DnsExchange assumes that the underlying stream is responsible for this.
84#[must_use = "futures do nothing unless polled"]
85pub struct DnsExchange<P> {
86    sender: BufDnsRequestStreamHandle<P>,
87}
88
89impl<P: RuntimeProvider> DnsExchange<P> {
90    /// Initializes a TcpStream with an existing tcp::TcpStream.
91    ///
92    /// This is intended for use with a TcpListener and Incoming.
93    ///
94    /// # Arguments
95    ///
96    /// * `stream` - the established IO stream for communication
97    pub fn from_stream<S>(stream: S) -> (Self, DnsExchangeBackground<S, P::Timer>)
98    where
99        S: DnsRequestSender + 'static + Send + Unpin,
100    {
101        let (sender, outbound_messages) = mpsc::channel(CHANNEL_BUFFER_SIZE);
102        let message_sender = BufDnsRequestStreamHandle {
103            sender,
104            _phantom: PhantomData,
105        };
106
107        Self::from_stream_with_receiver(stream, outbound_messages, message_sender)
108    }
109
110    /// Wraps a stream where a sender and receiver have already been established
111    pub fn from_stream_with_receiver<S>(
112        stream: S,
113        receiver: mpsc::Receiver<OneshotDnsRequest>,
114        sender: BufDnsRequestStreamHandle<P>,
115    ) -> (Self, DnsExchangeBackground<S, P::Timer>)
116    where
117        S: DnsRequestSender + 'static + Send + Unpin,
118    {
119        let background = DnsExchangeBackground {
120            io_stream: stream,
121            outbound_messages: receiver.peekable(),
122            marker: PhantomData,
123        };
124
125        (Self { sender }, background)
126    }
127
128    /// Returns a future, which itself wraps a future which is awaiting connection.
129    ///
130    /// The connect_future should be lazy.
131    #[cfg(feature = "std")]
132    pub fn connect<F, S>(connect_future: F) -> DnsExchangeConnect<F, S, P>
133    where
134        F: Future<Output = Result<S, io::Error>> + 'static + Send + Unpin,
135        S: DnsRequestSender + 'static + Send + Unpin,
136    {
137        let (sender, outbound_messages) = mpsc::channel(CHANNEL_BUFFER_SIZE);
138        let message_sender = BufDnsRequestStreamHandle {
139            sender,
140            _phantom: PhantomData,
141        };
142
143        DnsExchangeConnect::connect(connect_future, outbound_messages, message_sender)
144    }
145
146    /// Returns a future that returns an error immediately.
147    pub fn error<F, S>(error: io::Error) -> DnsExchangeConnect<F, S, P>
148    where
149        F: Future<Output = Result<S, io::Error>> + 'static + Send + Unpin,
150        S: DnsRequestSender + 'static + Send + Unpin,
151    {
152        DnsExchangeConnect(DnsExchangeConnectInner::Error(error))
153    }
154}
155
156impl<P: Clone> Clone for DnsExchange<P> {
157    fn clone(&self) -> Self {
158        Self {
159            sender: self.sender.clone(),
160        }
161    }
162}

177/// A Stream that will resolve to Responses after sending the request
178#[must_use = "futures do nothing unless polled"]
179pub struct DnsExchangeSend<P> {
180    result: DnsResponseReceiver,
181    _sender: BufDnsRequestStreamHandle<P>,
182}
183
184impl<P: Unpin> Stream for DnsExchangeSend<P> {
185    type Item = Result<DnsResponse, ProtoError>;
186
187    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
188        // as long as there is no result, poll the exchange
189        self.result.poll_next_unpin(cx)
190    }
191}

193/// This background future is responsible for driving all network operations for the DNS protocol.
194///
195/// It must be spawned before any DNS messages are sent.
196#[must_use = "futures do nothing unless polled"]
197pub struct DnsExchangeBackground<S, TE>
198where
199    S: DnsRequestSender + 'static + Send + Unpin,
200{
201    io_stream: S,
202    outbound_messages: Peekable<mpsc::Receiver<OneshotDnsRequest>>,
203    marker: PhantomData<TE>,
204}
205
206impl<S, TE> DnsExchangeBackground<S, TE>
207where
208    S: DnsRequestSender + 'static + Send + Unpin,
209{
210    fn pollable_split(&mut self) -> (&mut S, &mut Peekable<mpsc::Receiver<OneshotDnsRequest>>) {
211        (&mut self.io_stream, &mut self.outbound_messages)
212    }
213}
214
215impl<S, TE> Future for DnsExchangeBackground<S, TE>
216where
217    S: DnsRequestSender + 'static + Send + Unpin,
218    TE: Time + Unpin,
219{
220    type Output = Result<(), ProtoError>;
221
222    #[allow(clippy::unused_unit)]
223    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
224        let (io_stream, outbound_messages) = self.pollable_split();
225        let mut io_stream = Pin::new(io_stream);
226        let mut outbound_messages = Pin::new(outbound_messages);
227
228        // this will not accept incoming data while there is data to send
229        //  makes this self throttling.
230        loop {
231            // poll the underlying stream, to drive it...
232            match io_stream.as_mut().poll_next(cx) {
233                // The stream is ready
234                Poll::Ready(Some(Ok(()))) => (),
235                Poll::Pending => {
236                    if io_stream.is_shutdown() {
237                        // the io_stream is in a shutdown state, we are only waiting for final results...
238                        return Poll::Pending;
239                    }
240
241                    // NotReady and not shutdown, see if there are more messages to send
242                    ()
243                } // underlying stream is complete.
244                Poll::Ready(None) => {
245                    debug!("io_stream is done, shutting down");
246                    // TODO: return shutdown error to anything in the stream?
247
248                    return Poll::Ready(Ok(()));
249                }
250                Poll::Ready(Some(Err(error))) => {
251                    debug!(
252                        %error,
253                        "io_stream hit an error, shutting down"
254                    );
255
256                    return Poll::Ready(Err(error));
257                }
258            }
259
260            // then see if there is more to send
261            match outbound_messages.as_mut().poll_next(cx) {
262                // already handled above, here to make sure the poll() pops the next message
263                Poll::Ready(Some(dns_request)) => {
264                    // if there is no peer, this connection should die...
265                    let (dns_request, serial_response): (DnsRequest, _) = dns_request.into_parts();
266
267                    // Try to forward the `DnsResponseStream` to the requesting task. If we fail,
268                    // it must be because the requesting task has gone away / is no longer
269                    // interested. In that case, we can just log a warning, but there's no need
270                    // to take any more serious measures (such as shutting down this task).
271                    match serial_response.send_response(io_stream.send_message(dns_request)) {
272                        Ok(()) => (),
273                        Err(_) => {
274                            debug!("failed to associate send_message response to the sender");
275                        }
276                    }
277                }
278                // On not ready, this is our time to return...
279                Poll::Pending => return Poll::Pending,
280                Poll::Ready(None) => {
281                    // if there is nothing that can use this connection to send messages, then this is done...
282                    io_stream.shutdown();
283
284                    // now we'll await the stream to shutdown... see io_stream poll above
285                }
286            }
287
288            // else we loop to poll on the outbound_messages
289        }
290    }
291}

293/// A wrapper for a future DnsExchange connection.
294///
295/// DnsExchangeConnect is cloneable, making it possible to share this if the connection
296///  will be shared across threads.
297///
298/// The future will return a tuple of the DnsExchange (for sending messages) and a background
299///  for running the background tasks. The background is optional as only one thread should run
300///  the background. If returned, it must be spawned before any dns requests will function.
301pub struct DnsExchangeConnect<F, S, P>(DnsExchangeConnectInner<F, S, P>)
302where
303    F: Future<Output = Result<S, io::Error>> + 'static + Send + Unpin,
304    S: DnsRequestSender + 'static,
305    P: RuntimeProvider;
306
307impl<F, S, P> DnsExchangeConnect<F, S, P>
308where
309    F: Future<Output = Result<S, io::Error>> + 'static + Send + Unpin,
310    S: DnsRequestSender + 'static,
311    P: RuntimeProvider,
312{
313    fn connect(
314        connect_future: F,
315        outbound_messages: mpsc::Receiver<OneshotDnsRequest>,
316        sender: BufDnsRequestStreamHandle<P>,
317    ) -> Self {
318        Self(DnsExchangeConnectInner::Connecting {
319            connect_future,
320            outbound_messages: Some(outbound_messages),
321            sender: Some(sender),
322        })
323    }
324}
325
326impl<F, S, P> Future for DnsExchangeConnect<F, S, P>
327where
328    F: Future<Output = Result<S, io::Error>> + 'static + Send + Unpin,
329    S: DnsRequestSender + 'static + Send + Unpin,
330    P: RuntimeProvider,
331{
332    type Output = Result<(DnsExchange<P>, DnsExchangeBackground<S, P::Timer>), io::Error>;
333
334    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
335        self.0.poll_unpin(cx)
336    }
337}

339#[allow(clippy::large_enum_variant)]
340enum DnsExchangeConnectInner<F, S, P>
341where
342    F: Future<Output = Result<S, io::Error>> + 'static + Send,
343    S: DnsRequestSender + 'static + Send,
344    P: RuntimeProvider,
345{
346    Connecting {
347        connect_future: F,
348        outbound_messages: Option<mpsc::Receiver<OneshotDnsRequest>>,
349        sender: Option<BufDnsRequestStreamHandle<P>>,
350    },
351    Connected {
352        exchange: DnsExchange<P>,
353        background: Option<DnsExchangeBackground<S, P::Timer>>,
354    },
355    FailAll {
356        error: io::Error,
357        outbound_messages: mpsc::Receiver<OneshotDnsRequest>,
358    },
359    Error(io::Error),
360}
361
362impl<F, S, P> Future for DnsExchangeConnectInner<F, S, P>
363where
364    F: Future<Output = Result<S, io::Error>> + 'static + Send + Unpin,
365    S: DnsRequestSender + 'static + Send + Unpin,
366    P: RuntimeProvider,
367{
368    type Output = Result<(DnsExchange<P>, DnsExchangeBackground<S, P::Timer>), io::Error>;
369
370    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
371        loop {
372            let next;
373            match &mut *self {
374                Self::Connecting {
375                    connect_future,
376                    outbound_messages,
377                    sender,
378                } => {
379                    let connect_future = Pin::new(connect_future);
380                    match connect_future.poll(cx) {
381                        Poll::Ready(Ok(stream)) => {
382                            //debug!("connection established: {}", stream);
383
384                            let (exchange, background) = DnsExchange::from_stream_with_receiver(
385                                stream,
386                                outbound_messages
387                                    .take()
388                                    .expect("cannot poll after complete"),
389                                sender.take().expect("cannot poll after complete"),
390                            );
391
392                            next = Self::Connected {
393                                exchange,
394                                background: Some(background),
395                            };
396                        }
397                        Poll::Pending => return Poll::Pending,
398                        Poll::Ready(Err(error)) => {
399                            debug!(%error, "stream errored while connecting");
400                            next = Self::FailAll {
401                                error,
402                                outbound_messages: outbound_messages
403                                    .take()
404                                    .expect("cannot poll after complete"),
405                            }
406                        }
407                    };
408                }
409                Self::Connected {
410                    exchange,
411                    background,
412                } => {
413                    let exchange = exchange.clone();
414                    let background = background.take().expect("cannot poll after complete");
415
416                    return Poll::Ready(Ok((exchange, background)));
417                }
418                Self::FailAll {
419                    error,
420                    outbound_messages,
421                } => {
422                    while let Some(outbound_message) = match outbound_messages.poll_next_unpin(cx) {
423                        Poll::Ready(opt) => opt,
424                        Poll::Pending => return Poll::Pending,
425                    } {
426                        // ignoring errors... best effort send...
427                        let error =
428                            ProtoError::from(io::Error::new(error.kind(), error.to_string()));
429                        let _ = outbound_message
430                            .into_parts()
431                            .1
432                            .send_response(error.clone().into());
433                    }
434
435                    return Poll::Ready(Err(mem::replace(error, io::Error::other("taken"))));
436                }
437                Self::Error(error) => {
438                    return Poll::Ready(Err(mem::replace(error, io::Error::other("taken"))));
439                }
440            }
441
442            *self = next;
443        }
444    }

16/// [RFC 8162](https://datatracker.ietf.org/doc/html/rfc8162#section-2)
17///
18/// > The SMIMEA wire format and presentation format are the same as for
19/// > the [TLSA] record
20#[cfg_attr(feature = "serde", derive(Deserialize, Serialize))]
21#[derive(Debug, PartialEq, Eq, Hash, Clone)]
22pub struct SMIMEA(TLSA);
23
24impl SMIMEA {
25    /// Construct a new SMIMEA
26    ///
27    /// [RFC 8162](https://datatracker.ietf.org/doc/html/rfc8162#section-2)
28    ///
29    /// ```text
30    /// 2.  The SMIMEA Resource Record
31    ///
32    ///    The SMIMEA DNS resource record (RR) is used to associate an end
33    ///    entity certificate or public key with the associated email address,
34    ///    thus forming a "SMIMEA certificate association".  The semantics of
35    ///    how the SMIMEA resource record is interpreted are given later in this
36    ///    document.  Note that the information returned in the SMIMEA record
37    ///    might be for the end entity certificate, or it might be for the trust
38    ///    anchor or an intermediate certificate.  This mechanism is similar to
39    ///    the one given in [RFC7929] for OpenPGP.
40    ///
41    ///    The type value for the SMIMEA RRtype is defined in Section 8.  The
42    ///    SMIMEA resource record is class independent.
43    /// ```
44    pub fn new(
45        cert_usage: CertUsage,
46        selector: Selector,
47        matching: Matching,
48        cert_data: Vec<u8>,
49    ) -> Self {
50        Self(TLSA::new(cert_usage, selector, matching, cert_data))
51    }
52}
53
54/// This implementation allows calling the associated functions of [TLSA] on [SMIMEA].
55/// Since they contain exactly the same data, duplicating them would be pointless.
56impl Deref for SMIMEA {
57    type Target = TLSA;
58
59    fn deref(&self) -> &Self::Target {
60        &self.0
61    }
62}

68    }

71impl RecordDataDecodable<'_> for SMIMEA {

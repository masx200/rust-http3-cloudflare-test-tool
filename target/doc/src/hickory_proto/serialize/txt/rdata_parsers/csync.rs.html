8//! CSYNC record for synchronizing information to the parent zone
9
10use alloc::collections::BTreeSet;
11use alloc::string::ToString;
12use core::str::FromStr;
13
14use crate::rr::RecordType;
15use crate::rr::rdata::CSYNC;
16use crate::serialize::txt::errors::{ParseError, ParseErrorKind, ParseResult};
17
18/// Parse the RData from a set of Tokens
19///
20/// ```text
21/// IN CSYNC 1 3 A NS AAAA
22/// IN CSYNC 66 0 MX
23/// ```
24pub(crate) fn parse<'i, I: Iterator<Item = &'i str>>(mut tokens: I) -> ParseResult<CSYNC> {
25    let soa_serial: u32 = tokens
26        .next()
27        .ok_or_else(|| ParseError::from(ParseErrorKind::MissingToken("soa_serial".to_string())))
28        .and_then(|s| s.parse().map_err(Into::into))?;
29
30    let flags: u16 = tokens
31        .next()
32        .ok_or_else(|| ParseError::from(ParseErrorKind::MissingToken("flags".to_string())))
33        .and_then(|s| s.parse().map_err(Into::into))?;
34
35    let immediate: bool = flags & 0b0000_0001 == 0b0000_0001;
36    let soa_minimum: bool = flags & 0b0000_0010 == 0b0000_0010;
37
38    let mut record_types = BTreeSet::new();
39
40    for token in tokens {
41        record_types.insert(RecordType::from_str(token)?);
42    }
43
44    Ok(CSYNC::new(soa_serial, immediate, soa_minimum, record_types))
45}
46
47#[test]
48fn test_parsing() {
49    // IN CSYNC 123 3 NS
50
51    assert_eq!(
52        parse(vec!["123", "3", "NS"].into_iter()).expect("failed to parse CSYNC"),
53        CSYNC::new(123, true, true, [RecordType::NS]),
54    );
55}
56
57#[test]
58fn test_parsing_fails() {
59    // IN CSYNC NS
60
61    assert!(parse(vec!["NS"].into_iter()).is_err());
62    assert!(parse(vec![].into_iter()).is_err());
63}

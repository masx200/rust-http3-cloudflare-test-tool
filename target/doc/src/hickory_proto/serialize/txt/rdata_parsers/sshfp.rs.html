8//! SSHFP records for SSH public key fingerprints

16/// Parse the RData from a set of Tokens
17///
18/// [RFC 4255](https://tools.ietf.org/html/rfc4255#section-3.2)
19///
20/// ```text
21/// 3.2.  Presentation Format of the SSHFP RR
22///
23///    The RDATA of the presentation format of the SSHFP resource record
24///    consists of two numbers (algorithm and fingerprint type) followed by
25///    the fingerprint itself, presented in hex, e.g.:
26///
27///        host.example.  SSHFP 2 1 123456789abcdef67890123456789abcdef67890
28///
29///    The use of mnemonics instead of numbers is not allowed.
30/// ```
31pub(crate) fn parse<'i, I: Iterator<Item = &'i str>>(mut tokens: I) -> ParseResult<SSHFP> {
32    fn missing_field<E: From<ParseErrorKind>>(field: &str) -> E {
33        ParseErrorKind::Msg(format!("SSHFP {field} field missing")).into()
34    }
35    let (algorithm, fingerprint_type) = {
36        let mut parse_u8 = |field: &str| {
37            tokens
38                .next()
39                .ok_or_else(|| missing_field(field))
40                .and_then(|t| t.parse::<u8>().map_err(ParseError::from))
41        };
42        (
43            parse_u8("algorithm")?.into(),
44            parse_u8("fingerprint type")?.into(),
45        )
46    };
47    let fingerprint = sshfp::HEX.decode(
48        tokens
49            .next()
50            .filter(|fp| !fp.is_empty())
51            .ok_or_else(|| missing_field::<ParseError>("fingerprint"))?
52            .as_bytes(),
53    )?;
54    Some(SSHFP::new(algorithm, fingerprint_type, fingerprint))
55        .filter(|_| tokens.next().is_none())
56        .ok_or_else(|| ParseErrorKind::Message("too many fields for SSHFP").into())
57}

59#[test]
60fn test_parsing() {
61    assert!(parse(core::iter::empty()).is_err());
62    assert!(parse(vec!["51", "13"].into_iter()).is_err());
63    assert!(parse(vec!["1", "-1"].into_iter()).is_err());
64    assert!(parse(vec!["1", "1", "abcd", "foo"].into_iter()).is_err());
65
66    use crate::rr::rdata::sshfp::Algorithm::*;
67    use crate::rr::rdata::sshfp::FingerprintType::*;
68    use crate::rr::rdata::sshfp::{Algorithm, FingerprintType};
69
70    fn test_parsing(input: Vec<&str>, a: Algorithm, ft: FingerprintType, f: &[u8]) {
71        assert!(
72            parse(input.into_iter())
73                .map(|rd| rd == SSHFP::new(a, ft, f.to_vec()))
74                .unwrap_or(false)
75        );
76    }

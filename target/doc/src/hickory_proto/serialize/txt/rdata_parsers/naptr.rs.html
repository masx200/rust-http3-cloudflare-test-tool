11use core::str::FromStr;

13use crate::rr::Name;
14use crate::rr::rdata::naptr::{NAPTR, verify_flags};
15use crate::serialize::txt::errors::{ParseError, ParseErrorKind, ParseResult};

17/// Parse the RData from a set of Tokens
18///
19/// ```text
20/// ;;      order pflags service           regexp replacement
21/// IN NAPTR 100  50  "a"    "z3950+N2L+N2C"     ""   cidserver.example.com.
22/// IN NAPTR 100  50  "a"    "rcds+N2C"          ""   cidserver.example.com.
23/// IN NAPTR 100  50  "s"    "http+N2L+N2C+N2R"  ""   www.example.com.
24/// ```
25pub(crate) fn parse<'i, I: Iterator<Item = &'i str>>(
26    mut tokens: I,
27    origin: Option<&Name>,
28) -> ParseResult<NAPTR> {
29    let order: u16 = tokens
30        .next()
31        .ok_or_else(|| ParseError::from(ParseErrorKind::MissingToken("order".to_string())))
32        .and_then(|s| u16::from_str(s).map_err(Into::into))?;
33
34    let preference: u16 = tokens
35        .next()
36        .ok_or_else(|| ParseError::from(ParseErrorKind::MissingToken("preference".to_string())))
37        .and_then(|s| u16::from_str(s).map_err(Into::into))?;
38
39    let flags = tokens
40        .next()
41        .ok_or_else(|| ParseError::from(ParseErrorKind::MissingToken("flags".to_string())))
42        .map(ToString::to_string)
43        .map(|s| s.into_bytes().into_boxed_slice())?;
44    if !verify_flags(&flags) {
45        return Err(ParseError::from("bad flags, must be in range [a-zA-Z0-9]"));
46    }
47
48    let service = tokens
49        .next()
50        .ok_or_else(|| ParseError::from(ParseErrorKind::MissingToken("service".to_string())))
51        .map(ToString::to_string)
52        .map(|s| s.into_bytes().into_boxed_slice())?;
53
54    let regexp = tokens
55        .next()
56        .ok_or_else(|| ParseError::from(ParseErrorKind::MissingToken("regexp".to_string())))
57        .map(ToString::to_string)
58        .map(|s| s.into_bytes().into_boxed_slice())?;
59
60    let replacement: Name = tokens
61        .next()
62        .ok_or_else(|| ParseErrorKind::MissingToken("replacement".to_string()).into())
63        .and_then(|s| Name::parse(s, origin).map_err(ParseError::from))?;
64
65    Ok(NAPTR::new(
66        order,
67        preference,
68        flags,
69        service,
70        regexp,
71        replacement,
72    ))
73}

75#[test]
76fn test_parsing() {
77    // IN NAPTR 100  50  "a"    "z3950+N2L+N2C"     ""   cidserver.example.com.
78    // IN NAPTR 100  50  "a"    "rcds+N2C"          ""   cidserver.example.com.
79    // IN NAPTR 100  50  "s"    "http+N2L+N2C+N2R"  ""   www.example.com.
80
81    assert_eq!(
82        parse(
83            vec!["100", "50", "a", "z3950+N2L+N2C", "", "cidserver"].into_iter(),
84            Some(&Name::from_str("example.com.").unwrap())
85        )
86        .expect("failed to parse NAPTR"),
87        NAPTR::new(
88            100,
89            50,
90            b"a".to_vec().into_boxed_slice(),
91            b"z3950+N2L+N2C".to_vec().into_boxed_slice(),
92            b"".to_vec().into_boxed_slice(),
93            Name::from_str("cidserver.example.com.").unwrap()
94        ),
95    );
96}

98#[test]
99fn test_parsing_fails() {
100    // IN NAPTR 100  50  "a"    "z3950+N2L+N2C"     ""   cidserver.example.com.
101    // IN NAPTR 100  50  "a"    "rcds+N2C"          ""   cidserver.example.com.
102    // IN NAPTR 100  50  "s"    "http+N2L+N2C+N2R"  ""   www.example.com.
103
104    assert!(
105        parse(
106            vec!["100", "50", "-", "z3950+N2L+N2C", "", "cidserver"].into_iter(),
107            Some(&Name::from_str("example.com.").unwrap())
108        )
109        .is_err()
110    );

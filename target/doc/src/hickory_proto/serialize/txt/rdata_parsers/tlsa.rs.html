14use crate::rr::rdata::tlsa::{CertUsage, Matching, Selector};

16use crate::serialize::txt::errors::{ParseError, ParseErrorKind, ParseResult};

18fn to_u8(data: &str) -> ParseResult<u8> {
19    data.parse().map_err(ParseError::from)
20}
21
22/// Parse the RData from a set of Tokens
23pub(crate) fn parse<'i, I: Iterator<Item = &'i str>>(tokens: I) -> ParseResult<TLSA> {
24    parse_impl(tokens).map(|(usage, selector, matching, cert_data)| {
25        TLSA::new(usage, selector, matching, cert_data)
26    })
27}
28
29/// [RFC 6698, DNS-Based Authentication for TLS](https://tools.ietf.org/html/rfc6698#section-2.2)
30///
31/// ```text
32/// 2.2.  TLSA RR Presentation Format
33///
34///    The presentation format of the RDATA portion (as defined in
35///    [RFC1035]) is as follows:
36///
37///    o  The certificate usage field MUST be represented as an 8-bit
38///       unsigned integer.
39///
40///    o  The selector field MUST be represented as an 8-bit unsigned
41///       integer.
42///
43///    o  The matching type field MUST be represented as an 8-bit unsigned
44///       integer.
45///
46///    o  The certificate association data field MUST be represented as a
47///       string of hexadecimal characters.  Whitespace is allowed within
48///       the string of hexadecimal characters, as described in [RFC1035].
49/// ```
50pub(super) fn parse_impl<'i, I: Iterator<Item = &'i str>>(
51    tokens: I,
52) -> ParseResult<(CertUsage, Selector, Matching, Vec<u8>)> {
53    let mut iter = tokens;
54
55    let token: &str = iter
56        .next()
57        .ok_or_else(|| ParseError::from(ParseErrorKind::Message("TLSA usage field missing")))?;
58    let usage = CertUsage::from(to_u8(token)?);
59
60    let token = iter
61        .next()
62        .ok_or(ParseErrorKind::Message("TLSA selector field missing"))?;
63    let selector = to_u8(token)?.into();
64
65    let token = iter
66        .next()
67        .ok_or(ParseErrorKind::Message("TLSA matching field missing"))?;
68    let matching = to_u8(token)?.into();
69
70    // these are all in hex: "a string of hexadecimal characters"
71    //   aside: personally I find it funny that the other fields are decimal, while this is hex encoded...
72    let cert_data = iter.fold(String::new(), |mut cert_data, data| {
73        cert_data.push_str(data);
74        cert_data
75    });
76    let cert_data = sshfp::HEX.decode(cert_data.as_bytes())?;
77
78    if !cert_data.is_empty() {
79        Ok((usage, selector, matching, cert_data))
80    } else {
81        Err(ParseErrorKind::Message("TLSA data field missing").into())
82    }
83}

27/// [RFC 9460 SVCB and HTTPS Resource Records, Nov 2023](https://datatracker.ietf.org/doc/html/rfc9460#section-2.1)
28///
29/// ```text
30/// 2.1.  Zone file presentation format
31///
32///   The presentation format <RDATA> of the record ([RFC1035]) has the form:
33///
34///   SvcPriority TargetName SvcParams
35///
36///   The SVCB record is defined specifically within the Internet ("IN")
37///   Class ([RFC1035]).
38///
39///   SvcPriority is a number in the range 0-65535, TargetName is a
40///   <domain-name> ([RFC1035], Section 5.1), and the SvcParams are
41///   a whitespace-separated list, with each SvcParam consisting of a
42///   SvcParamKey=SvcParamValue pair or a standalone SvcParamKey.
43///   SvcParamKeys are registered by IANA  (Section 14.3).
44///
45///   Each SvcParamKey SHALL appear at most once in the SvcParams.  In
46///   presentation format, SvcParamKeys are lowercase alphanumeric
47///   strings.  Key names should contain 1-63 characters from the ranges
48///   "a"-"z", "0"-"9", and "-".  In ABNF [RFC5234],
49///
50///   alpha-lc      = %x61-7A   ;  a-z
51///   SvcParamKey   = 1*63(alpha-lc / DIGIT / "-")
52///   SvcParam      = SvcParamKey ["=" SvcParamValue]
53///   SvcParamValue = char-string ; See Appendix A.
54///   value         = *OCTET ; Value before key-specific parsing
55///
56///   The SvcParamValue is parsed using the character-string decoding
57///   algorithm (Appendix A), producing a value.  The value is then
58///   validated and converted into wire-format in a manner specific to each
59///   key.
60///
61///   When the optional "=" and SvcParamValue are omitted, the value is
62///   interpreted as empty.
63///
64///   Arbitrary keys can be represented using the unknown-key presentation
65///   format "keyNNNNN" where NNNNN is the numeric value of the key type
66///   without leading zeros. A SvcParam in this form SHALL be parsed as
67///   specified above, and the decoded value SHALL be used as its wire-format
68///   encoding.
69///
70///   For some SvcParamKeys, the value corresponds to a list or set of
71///   items.  Presentation formats for such keys SHOULD use a comma-
72///   separated list (Appendix A.1).
73///
74///   SvcParams in presentation format MAY appear in any order, but keys
75///   MUST NOT be repeated.
76/// ```
77pub(crate) fn parse<'i, I: Iterator<Item = &'i str>>(mut tokens: I) -> ParseResult<SVCB> {
78    // SvcPriority
79    let svc_priority: u16 = tokens
80        .next()
81        .ok_or_else(|| ParseError::from(ParseErrorKind::MissingToken("SvcPriority".to_string())))
82        .and_then(|s| s.parse().map_err(Into::into))?;
83
84    // svcb target
85    let target_name: Name = tokens
86        .next()
87        .ok_or_else(|| ParseError::from(ParseErrorKind::MissingToken("Target".to_string())))
88        .and_then(|s| Name::from_str(s).map_err(ParseError::from))?;
89
90    // Loop over all of the service parameters
91    let mut svc_params = Vec::new();
92    for token in tokens {
93        // first need to split the key and (optional) value
94        let mut key_value = token.splitn(2, '=');
95        let key = key_value.next().ok_or_else(|| {
96            ParseError::from(ParseErrorKind::MissingToken(
97                "SVCB SvcbParams missing".to_string(),
98            ))
99        })?;
100
101        // get the value, and remove any quotes
102        let mut value = key_value.next();
103        if let Some(value) = value.as_mut() {
104            if value.starts_with('"') && value.ends_with('"') {
105                *value = &value[1..value.len() - 1];
106            }
107        }
108        svc_params.push(into_svc_param(key, value)?);
109    }
110
111    Ok(SVCB::new(svc_priority, target_name, svc_params))
112}

114// first take the param and convert to
115fn into_svc_param(
116    key: &str,
117    value: Option<&str>,
118) -> Result<(SvcParamKey, SvcParamValue), ParseError> {
119    let key = SvcParamKey::from_str(key)?;
120    let value = parse_value(key, value)?;
121
122    Ok((key, value))
123}
124
125fn parse_value(key: SvcParamKey, value: Option<&str>) -> Result<SvcParamValue, ParseError> {
126    match key {
127        SvcParamKey::Mandatory => parse_mandatory(value),
128        SvcParamKey::Alpn => parse_alpn(value),
129        SvcParamKey::NoDefaultAlpn => parse_no_default_alpn(value),
130        SvcParamKey::Port => parse_port(value),
131        SvcParamKey::Ipv4Hint => parse_ipv4_hint(value),
132        SvcParamKey::Ipv6Hint => parse_ipv6_hint(value),
133        SvcParamKey::EchConfigList => parse_ech_config(value),
134        SvcParamKey::Key(_) => parse_unknown(value),
135        SvcParamKey::Key65535 | SvcParamKey::Unknown(_) => {
136            Err(ParseError::from(ParseErrorKind::Message(
137                "Bad Key type or unsupported, see generic key option, e.g. key1234",
138            )))
139        }
140    }
141}

143fn parse_char_data(value: &str) -> Result<String, ParseError> {
144    let mut lex = Lexer::new(value);
145    let ch_data = lex
146        .next_token()?
147        .ok_or_else(|| ParseError::from(ParseErrorKind::Message("expected character data")))?;
148
149    match ch_data {
150        Token::CharData(data) => Ok(data),
151        _ => Err(ParseError::from(ParseErrorKind::Message(
152            "expected character data",
153        ))),
154    }
155}

157/// [RFC 9460 SVCB and HTTPS Resource Records, Nov 2023](https://datatracker.ietf.org/doc/html/rfc9460#section-8)
158///
159/// ```text
160///   The presentation value SHALL be a comma-separated list
161///   (Appendix A.1) of one or more valid SvcParamKeys, either by their
162///   registered name or in the unknown-key format (Section 2.1).  Keys MAY
163///   appear in any order, but MUST NOT appear more than once.  For self-
164///   consistency (Section 2.4.3), listed keys MUST also appear in the
165///   SvcParams.
166///
167///   To enable simpler parsing, this SvcParamValue MUST NOT contain escape
168///   sequences.
169///
170///   For example, the following is a valid list of SvcParams:
171///
172///   ipv6hint=... key65333=ex1 key65444=ex2 mandatory=key65444,ipv6hint
173/// ```
174///
175/// Currently this does not validate that the mandatory section matches the other keys
176fn parse_mandatory(value: Option<&str>) -> Result<SvcParamValue, ParseError> {
177    let value = value.ok_or_else(|| {
178        ParseError::from(ParseErrorKind::Message(
179            "expected at least one Mandatory field",
180        ))
181    })?;
182
183    let mandatories = parse_list::<SvcParamKey>(value)?;
184    Ok(SvcParamValue::Mandatory(Mandatory(mandatories)))
185}

187/// [RFC 9460 SVCB and HTTPS Resource Records, Nov 2023](https://datatracker.ietf.org/doc/html/rfc9460#section-7.1.1)
188///
189/// ```text
190///   ALPNs are identified by their registered "Identification Sequence"
191///   (alpn-id), which is a sequence of 1-255 octets.
192///
193///   alpn-id = 1*255OCTET
194///
195///   For "alpn", the presentation value SHALL be a comma-separated list
196///   (Appendix A.1) of one or more alpn-ids.
197/// ```
198///
199/// This does not currently check to see if the ALPN code is legitimate
200fn parse_alpn(value: Option<&str>) -> Result<SvcParamValue, ParseError> {
201    let value = value.ok_or_else(|| {
202        ParseError::from(ParseErrorKind::Message("expected at least one ALPN code"))
203    })?;
204
205    let alpns = parse_list::<String>(value).expect("infallible");
206    Ok(SvcParamValue::Alpn(Alpn(alpns)))
207}

209/// [RFC 9460 SVCB and HTTPS Resource Records, Nov 2023](https://datatracker.ietf.org/doc/html/rfc9460#section-7.1.1)
210///
211/// ```text
212///   For "no-default-alpn", the presentation and wire format values MUST
213///   be empty.  When "no-default-alpn" is specified in an RR, "alpn" must
214///   also be specified in order for the RR to be "self-consistent"
215///   (Section 2.4.3).
216/// ```
217fn parse_no_default_alpn(value: Option<&str>) -> Result<SvcParamValue, ParseError> {
218    if value.is_some() {
219        return Err(ParseErrorKind::Message("no value expected for NoDefaultAlpn").into());
220    }
221
222    Ok(SvcParamValue::NoDefaultAlpn)
223}

225/// [RFC 9460 SVCB and HTTPS Resource Records, Nov 2023](https://datatracker.ietf.org/doc/html/rfc9460#section-7.2)
226///
227/// ```text
228///   The presentation value of the SvcParamValue is a single decimal
229///   integer between 0 and 65535 in ASCII.  Any other value (e.g. an
230///   empty value) is a syntax error.  To enable simpler parsing, this
231///   SvcParam MUST NOT contain escape sequences.
232/// ```
233fn parse_port(value: Option<&str>) -> Result<SvcParamValue, ParseError> {
234    let value = value.ok_or_else(|| {
235        ParseError::from(ParseErrorKind::Message("a port number for the port option"))
236    })?;
237
238    let value = parse_char_data(value)?;
239    let port = u16::from_str(&value)?;
240    Ok(SvcParamValue::Port(port))
241}

277/// As the documentation states, the presentation format (what this function outputs) must be a BASE64 encoded string.
278///   hickory-dns will encode to BASE64 during formatting of the internal data, and output the BASE64 value.
279///
280/// [draft-ietf-tls-svcb-ech-01 Bootstrapping TLS Encrypted ClientHello with DNS Service Bindings, Sep 2024](https://datatracker.ietf.org/doc/html/draft-ietf-tls-svcb-ech-01)
281/// ```text
282///  In presentation format, the value is the ECHConfigList in Base 64 Encoding
283///  (Section 4 of [RFC4648]). Base 64 is used here to simplify integration with
284///  TLS server software. To enable simpler parsing, this SvcParam MUST NOT
285///  contain escape sequences.
286/// ```
287fn parse_ech_config(value: Option<&str>) -> Result<SvcParamValue, ParseError> {
288    let value = value.ok_or_else(|| {
289        ParseError::from(ParseErrorKind::Message(
290            "expected a base64 encoded string for EchConfig",
291        ))
292    })?;
293
294    let value = parse_char_data(value)?;
295    let ech_config_bytes = data_encoding::BASE64.decode(value.as_bytes())?;
296    Ok(SvcParamValue::EchConfigList(EchConfigList(
297        ech_config_bytes,
298    )))
299}

301///  [RFC 9460 SVCB and HTTPS Resource Records, Nov 2023](https://datatracker.ietf.org/doc/html/rfc9460#section-2.1)
302///
303/// ```text
304///   Arbitrary keys can be represented using the unknown-key presentation
305///   format "keyNNNNN" where NNNNN is the numeric value of the key type
306///   without leading zeros. A SvcParam in this form SHALL be parsed as specified
307///   above, and the decoded value SHALL be used as its wire-format encoding.
308///
309///   For some SvcParamKeys, the value corresponds to a list or set of
310///   items.  Presentation formats for such keys SHOULD use a comma-
311///   separated list (Appendix A.1).
312///
313///   SvcParams in presentation format MAY appear in any order, but keys
314///   MUST NOT be repeated.
315/// ```
316fn parse_unknown(value: Option<&str>) -> Result<SvcParamValue, ParseError> {
317    let unknown: Vec<u8> = if let Some(value) = value {
318        value.as_bytes().to_vec()
319    } else {
320        Vec::new()
321    };
322
323    Ok(SvcParamValue::Unknown(Unknown(unknown)))
324}

326fn parse_list<T>(value: &str) -> Result<Vec<T>, ParseError>
327where
328    T: FromStr,
329    T::Err: Into<ParseError>,
330{
331    let mut result = Vec::new();
332    let mut current_value = String::new();
333    let mut escaping = false;
334
335    for c in value.chars() {
336        match (c, escaping) {
337            // End of value
338            (',', false) => {
339                result.push(T::from_str(&parse_char_data(&current_value)?).map_err(Into::into)?);
340                current_value.clear()
341            }
342            // Start of escape sequence
343            ('\\', false) => escaping = true,
344            // Comma inside escape sequence
345            (',', true) => {
346                current_value.push(',');
347                escaping = false
348            }
349            // Regular character inside escape sequence
350            (_, true) => {
351                current_value.push(c);
352                escaping = false
353            }
354            // Regular character
355            (_, false) => current_value.push(c),
356        }
357    }
358
359    // Push the remaining value if there's any
360    if !current_value.is_empty() {
361        result.push(T::from_str(&parse_char_data(&current_value)?).map_err(Into::into)?);
362    }
363
364    Ok(result)
365}

381    fn parse_record<D: RecordData>(txt: &str) -> D {
382        let records = Parser::new(txt, None, Some(Name::root()))
383            .parse()
384            .expect("failed to parse record")
385            .1;
386        let record_set = records.into_iter().next().expect("no record found").1;
387        D::try_borrow(record_set.into_iter().next().unwrap().data())
388            .expect("Not the correct record")
389            .clone()
390    }

29/// A UDP client stream of DNS binary packets.
30///
31/// It is expected that the resolver wrapper will be responsible for creating and managing a new UDP
32/// client stream such that each request would have a random port. This is to avoid potential cache
33/// poisoning due to UDP spoofing attacks.
34#[must_use = "futures do nothing unless polled"]
35pub struct UdpClientStream<P> {
36    name_server: SocketAddr,
37    timeout: Duration,
38    is_shutdown: bool,
39    signer: Option<Arc<dyn MessageSigner>>,
40    bind_addr: Option<SocketAddr>,
41    avoid_local_ports: Arc<HashSet<u16>>,
42    os_port_selection: bool,
43    provider: P,
44    max_retries: u8,
45    retry_interval_floor: Duration,
46}
47
48impl<P: RuntimeProvider> UdpClientStream<P> {
49    /// Construct a new [`UdpClientStream`] via a [`UdpClientStreamBuilder`].
50    pub fn builder(name_server: SocketAddr, provider: P) -> UdpClientStreamBuilder<P> {
51        UdpClientStreamBuilder {
52            name_server,
53            timeout: None,
54            signer: None,
55            bind_addr: None,
56            avoid_local_ports: Arc::default(),
57            os_port_selection: false,
58            provider,
59            max_retries: 3,
60            // This is the default value to use for the retry interval floor, which acts as a lower
61            // bound on the retry interval.
62            retry_interval_floor: DEFAULT_RETRY_FLOOR,
63        }
64    }
65}

73impl<P: RuntimeProvider> DnsRequestSender for UdpClientStream<P> {
74    fn send_message(&mut self, request: DnsRequest) -> DnsResponseStream {
75        if self.is_shutdown {
76            panic!("can not send messages after stream is shutdown")
77        }
78
79        let retry_interval_time = request.options().retry_interval;
80        let request = UdpRequest::new(request, self);
81
82        let max_retries = self.max_retries;
83        let retry_interval = if retry_interval_time < self.retry_interval_floor {
84            self.retry_interval_floor
85        } else {
86            retry_interval_time
87        };
88
89        P::Timer::timeout(
90            self.timeout,
91            retry::<P>(request, retry_interval, max_retries.into()),
92        )
93        .into()
94    }
95
96    fn shutdown(&mut self) {
97        self.is_shutdown = true;
98    }
99
100    fn is_shutdown(&self) -> bool {
101        self.is_shutdown
102    }
103}

119/// Request context for data send_udp_message needs via the retry handler closure
120struct UdpRequest<P> {
121    avoid_local_ports: Arc<HashSet<u16>>,
122    name_server: SocketAddr,
123    request: DnsRequest,
124    provider: P,
125    signer: Option<Arc<dyn MessageSigner>>,
126    now: u64,
127    bind_addr: Option<SocketAddr>,
128    os_port_selection: bool,
129    case_randomization: bool,
130    recv_buf_size: usize,
131}
132
133impl<P: RuntimeProvider> UdpRequest<P> {
134    fn new(request: DnsRequest, stream: &UdpClientStream<P>) -> Self {
135        Self {
136            avoid_local_ports: stream.avoid_local_ports.clone(),
137            recv_buf_size: MAX_RECEIVE_BUFFER_SIZE.min(request.max_payload() as usize),
138            case_randomization: request.options().case_randomization,
139            name_server: stream.name_server,
140            // Only smuggle in the signer if we are going to use it.
141            signer: match &stream.signer {
142                Some(signer) if signer.should_sign_message(&request) => stream.signer.clone(),
143                _ => None,
144            },
145            request,
146            provider: stream.provider.clone(),
147            now: P::Timer::current_time(),
148            bind_addr: stream.bind_addr,
149            os_port_selection: stream.os_port_selection,
150        }
151    }
152}

154impl<P: RuntimeProvider> Request for UdpRequest<P> {
155    async fn send(&self) -> Result<DnsResponse, ProtoError> {
156        let original_query = self.request.original_query();
157        let mut request = self.request.clone();
158
159        let mut verifier = None;
160        if let Some(signer) = &self.signer {
161            match request.finalize(&**signer, self.now) {
162                Ok(answer_verifier) => verifier = answer_verifier,
163                Err(e) => {
164                    debug!("could not sign message: {}", e);
165                    return Err(e);
166                }
167            }
168        }
169
170        let request_bytes = match request.to_vec() {
171            Ok(bytes) => bytes,
172            Err(err) => return Err(err),
173        };
174
175        let msg_id = request.id();
176        let msg = SerialMessage::new(request_bytes, self.name_server);
177        let addr = msg.addr();
178        let final_message = match msg.to_message() {
179            Ok(m) => m,
180            Err(e) => return Err(e),
181        };
182        debug!(%final_message, "final message");
183
184        let socket = NextRandomUdpSocket::new(
185            addr,
186            self.bind_addr,
187            self.avoid_local_ports.clone(),
188            self.os_port_selection,
189            self.provider.clone(),
190        )
191        .await?;
192
193        let bytes = msg.bytes();
194        let len_sent: usize = socket.send_to(bytes, addr).await?;
195
196        if bytes.len() != len_sent {
197            return Err(ProtoError::from(format!(
198                "Not all bytes of message sent, {} of {}",
199                len_sent,
200                bytes.len()
201            )));
202        }
203
204        // Create the receive buffer.
205        trace!(
206            recv_buf_size = self.recv_buf_size,
207            "creating UDP receive buffer"
208        );
209        let mut recv_buf = vec![0; self.recv_buf_size];
210
211        // Try to process up to 3 responses
212        for _ in 0..3 {
213            let (len, src) = socket.recv_from(&mut recv_buf).await?;
214
215            // Copy the slice of read bytes.
216            let response_bytes = &recv_buf[0..len];
217            let response_buffer = Vec::from(response_bytes);
218
219            // compare expected src to received packet
220            let request_target = msg.addr();
221
222            // Comparing the IP and Port directly as internal information about the link is stored with the IpAddr, see https://github.com/hickory-dns/hickory-dns/issues/2081
223            if src.ip().to_canonical() != request_target.ip().to_canonical()
224                || src.port() != request_target.port()
225            {
226                warn!(
227                    "ignoring response from {}:{} because it does not match name_server: {}:{}.",
228                    src.ip().to_canonical(),
229                    src.port(),
230                    request_target.ip().to_canonical(),
231                    request_target.port(),
232                );
233
234                // await an answer from the correct NameServer
235                continue;
236            }
237
238            let mut response = match DnsResponse::from_buffer(response_buffer) {
239                Ok(response) => response,
240                Err(e) => {
241                    // on errors deserializing, continue
242                    warn!("dropped malformed message waiting for id: {msg_id} err: {e}");
243                    continue;
244                }
245            };
246
247            // Validate the message id in the response matches the value chosen for the query.
248            if msg_id != response.id() {
249                // on wrong id, attempted poison?
250                warn!(
251                    "expected message id: {} got: {}, dropped",
252                    msg_id,
253                    response.id()
254                );
255
256                return Err(ProtoErrorKind::BadTransactionId.into());
257            }
258
259            // Validate the returned query name.
260            //
261            // This currently checks that each response query name was present in the original query, but not that
262            // every original question is present.
263            //
264            // References:
265            //
266            // RFC 1035 7.3:
267            //
268            // The next step is to match the response to a current resolver request.
269            // The recommended strategy is to do a preliminary matching using the ID
270            // field in the domain header, and then to verify that the question section
271            // corresponds to the information currently desired.
272            //
273            // RFC 1035 7.4:
274            //
275            // In general, we expect a resolver to cache all data which it receives in
276            // responses since it may be useful in answering future client requests.
277            // However, there are several types of data which should not be cached:
278            //
279            // ...
280            //
281            //  - RR data in responses of dubious reliability.  When a resolver
282            // receives unsolicited responses or RR data other than that
283            // requested, it should discard it without caching it.
284            let request_message = Message::from_vec(msg.bytes())?;
285            let request_queries = request_message.queries();
286            let response_queries = response.queries_mut();
287
288            let question_matches = response_queries
289                .iter()
290                .all(|elem| request_queries.contains(elem));
291            if self.case_randomization
292                && question_matches
293                && !response_queries.iter().all(|elem| {
294                    request_queries
295                        .iter()
296                        .any(|req_q| req_q == elem && req_q.name().eq_case(elem.name()))
297                })
298            {
299                warn!(
300                    "case of question section did not match: we expected '{request_queries:?}', but received '{response_queries:?}' from server {src}"
301                );
302                return Err(ProtoErrorKind::QueryCaseMismatch.into());
303            }
304            if !question_matches {
305                warn!(
306                    "detected forged question section: we expected '{request_queries:?}', but received '{response_queries:?}' from server {src}"
307                );
308                continue;
309            }
310
311            // overwrite the query with the original query if case randomization may have been used
312            if self.case_randomization {
313                if let Some(original_query) = original_query {
314                    for response_query in response_queries.iter_mut() {
315                        if response_query == original_query {
316                            *response_query = original_query.clone();
317                        }
318                    }
319                }
320            }
321
322            debug!("received message id: {}", response.id());
323            if let Some(mut verifier) = verifier {
324                return verifier(response_bytes);
325            } else {
326                return Ok(response);
327            }
328        }
329
330        Err("udp receive attempts exceeded".into())
331    }
332}

456/// This implements a retry handler for tasks that might not complete successfully (e.g.,
457/// DNS requests made via UDP.) It starts a task future immediately, then every
458/// retry_interval_time period up to a maximum of max_tasks. It will immediately return
459/// the first task that completes successfully, or an error if no tasks succeed.
460/// It does not implement an overall timeout to bound the work.
461async fn retry<Provider: RuntimeProvider>(
462    request: impl Request,
463    retry_interval_time: Duration,
464    max_tasks: usize,
465) -> Result<DnsResponse, ProtoError> {
466    let mut futures = FuturesUnordered::new();
467
468    let retry_timer = Provider::Timer::delay_for(retry_interval_time).fuse();
469    pin_mut!(retry_timer);
470
471    futures.push(request.send());
472    let mut tasks = 1;
473
474    loop {
475        futures_util::select! {
476            result = futures.next() => {
477                match result {
478                    Some(result) => return result,
479                    None => return Err(ProtoError::from("no tasks successful")),
480                }
481            }
482            _ = &mut retry_timer => {
483                if tasks < max_tasks {
484                    tasks += 1;
485                    futures.push(request.send());
486                    retry_timer.set(Provider::Timer::delay_for(retry_interval_time).fuse());
487                }
488            }
489        }
490    }
491}

597        let (req, tries) = DelayedResponse::new(
598            DnsResponse::from_message(message.clone()).unwrap(),
599            Duration::from_millis(1500),

606        let (req, tries) = DelayedResponse::new(

608            Duration::from_millis(1000),
609            Arc::new(AtomicU8::new(0)),

638    struct DelayedResponse {
639        response: DnsResponse,
640        delay: Duration,
641        counter: Arc<AtomicU8>,
642    }
643
644    impl DelayedResponse {
645        fn new(
646            response: DnsResponse,
647            delay: Duration,
648            counter: Arc<AtomicU8>,
649        ) -> (Self, Arc<AtomicU8>) {
650            (
651                Self {
652                    response,
653                    delay,
654                    counter: counter.clone(),
655                },
656                counter,
657            )
658        }
659    }
660
661    impl Request for DelayedResponse {
662        async fn send(&self) -> Result<DnsResponse, ProtoError> {
663            let _ = self.counter.fetch_add(1, Ordering::Relaxed);
664            sleep(self.delay).await;
665            Ok(self.response.clone())
666        }
667    }

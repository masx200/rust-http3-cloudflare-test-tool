13/// Returns a value from the registry, and returns a default if it doesn't exist.
14fn get_value<T: FromRegValue>(
15    predef: HKEY,
16    subkey: &str,
17    value_name: &str,
18    default: T,
19) -> ::std::io::Result<T> {
20    let key = RegKey::predef(predef);
21
22    let value: T = match key.open_subkey_with_flags(subkey, KEY_READ) {
23        Ok(key) => match key.get_value(value_name) {
24            Ok(value) => value,
25            Err(err) => match err.kind() {
26                ::std::io::ErrorKind::NotFound => default,
27                _ => return Err(err),
28            },
29        },
30        Err(err) => match err.kind() {
31            ::std::io::ErrorKind::NotFound => default,
32            _ => return Err(err),
33        },
34    };
35
36    Ok(value)
37}

39const TCPIP_PARAMETERS_KEY_PATH: &str = "SYSTEM\\CurrentControlSet\\Services\\Tcpip\\Parameters";
40
41/// Returns the DNS suffix search list for the network connection used by the computer.
42pub fn get_search_list() -> Result<Vec<String>> {
43    let hklm = RegKey::predef(HKEY_LOCAL_MACHINE);
44    let params_key = hklm.open_subkey_with_flags(TCPIP_PARAMETERS_KEY_PATH, KEY_READ)?;
45    let search_list: ::std::io::Result<String> = params_key.get_value("SearchList");
46    if let Ok(search_list) = search_list {
47        let search_list: Vec<String> = search_list
48            .split(',')
49            .map(std::string::ToString::to_string)
50            .collect();
51        Ok(search_list)
52    } else {
53        Ok(vec![])
54    }
55}

57/// Returns the computer domain name (if any).
58/// Returns `None` if the computer does not belong to a domain.
59pub fn get_domain() -> Result<Option<String>> {
60    let hklm = RegKey::predef(HKEY_LOCAL_MACHINE);
61    let params_key = hklm.open_subkey_with_flags(TCPIP_PARAMETERS_KEY_PATH, KEY_READ)?;
62    let domain: String = params_key.get_value("Domain")?;
63
64    let domain = if domain.is_empty() {
65        None
66    } else {
67        Some(domain)
68    };
69    Ok(domain)
70}

72/// Returns `true` if the computer is configured to use the round robin strategy.
73/// Otherwise, returns `false`.
74pub fn is_round_robin_enabled() -> Result<bool> {
75    let rotate: u32 = get_value(
76        HKEY_LOCAL_MACHINE,
77        "SYSTEM\\CurrentControlSet\\Services\\DNS\\Parameters",
78        "RoundRobin",
79        1, // The default is 1 according to msdn
80    )?;
81    Ok(rotate != 0)
82}

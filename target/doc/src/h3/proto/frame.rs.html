78impl Frame<PayloadLen> {
79    pub const MAX_ENCODED_SIZE: usize = VarInt::MAX_SIZE * 7;
80
81    /// Decodes a Frame from the stream according to <https://www.rfc-editor.org/rfc/rfc9114#section-7.1>
82    pub fn decode<T: Buf>(buf: &mut T) -> Result<Self, FrameError> {
83        let remaining = buf.remaining();
84        let ty = FrameType::decode(buf).map_err(|_| FrameError::Incomplete(remaining + 1))?;
85
86        // Webtransport streams need special handling as they have no length.
87        //
88        // See: https://datatracker.ietf.org/doc/html/draft-ietf-webtrans-http3/#section-4.2
89        if ty == FrameType::WEBTRANSPORT_BI_STREAM {
90            #[cfg(feature = "tracing")]
91            tracing::trace!("webtransport frame");
92
93            return Ok(Frame::WebTransportStream(SessionId::decode(buf)?));
94        }
95
96        let len = buf
97            .get_var()
98            .map_err(|_| FrameError::Incomplete(remaining + 1))?;
99
100        if ty == FrameType::DATA {
101            return Ok(Frame::Data((len as usize).into()));
102        }
103
104        if buf.remaining() < len as usize {
105            return Err(FrameError::Incomplete(2 + len as usize));
106        }
107
108        let mut payload = buf.take(len as usize);
109
110        #[cfg(feature = "tracing")]
111        trace!("frame ty: {:?}", ty);
112
113        let frame = match ty {
114            FrameType::HEADERS => Ok(Frame::Headers(payload.copy_to_bytes(len as usize))),
115            FrameType::SETTINGS => Ok(Frame::Settings(Settings::decode(&mut payload)?)),
116            FrameType::CANCEL_PUSH => Ok(Frame::CancelPush(payload.get_var()?.try_into()?)),
117            FrameType::PUSH_PROMISE => Ok(Frame::PushPromise(PushPromise::decode(&mut payload)?)),
118            FrameType::GOAWAY => Ok(Frame::Goaway(VarInt::decode(&mut payload)?)),
119            FrameType::MAX_PUSH_ID => Ok(Frame::MaxPushId(payload.get_var()?.try_into()?)),
120            //= https://www.rfc-editor.org/rfc/rfc9114#section-7.2.8
121            //# These frame
122            //# types MUST NOT be sent, and their receipt MUST be treated as a
123            //# connection error of type H3_FRAME_UNEXPECTED.
124            FrameType::H2_PRIORITY
125            | FrameType::H2_PING
126            | FrameType::H2_WINDOW_UPDATE
127            | FrameType::H2_CONTINUATION => Err(FrameError::UnsupportedFrame(ty.0)),
128            FrameType::WEBTRANSPORT_BI_STREAM | FrameType::DATA => unreachable!(),
129            _ => {
130                buf.advance(len as usize);
131                //= https://www.rfc-editor.org/rfc/rfc9114#section-7.2.8
132                //# Endpoints MUST
133                //# NOT consider these frames to have any meaning upon receipt.
134                Err(FrameError::UnknownFrame(ty.0))
135            }
136        };
137
138        if let Ok(_frame) = &frame {
139            #[cfg(feature = "tracing")]
140            trace!(
141                "got frame {:?}, len: {}, remaining: {}",
142                _frame,
143                len,
144                buf.remaining()
145            );
146        }
147        frame
148    }
149}

151impl<B> Encode for Frame<B>
152where
153    B: Buf,
154{
155    fn encode<T: BufMut>(&self, buf: &mut T) {
156        match self {
157            Frame::Data(b) => {
158                FrameType::DATA.encode(buf);
159                buf.write_var(b.remaining() as u64);
160            }
161            Frame::Headers(f) => {
162                FrameType::HEADERS.encode(buf);
163                buf.write_var(f.len() as u64);
164            }
165            Frame::Settings(f) => f.encode(buf),
166            Frame::PushPromise(f) => f.encode(buf),
167            Frame::CancelPush(id) => simple_frame_encode(FrameType::CANCEL_PUSH, (*id).into(), buf),
168            Frame::Goaway(id) => simple_frame_encode(FrameType::GOAWAY, *id, buf),
169            Frame::MaxPushId(id) => simple_frame_encode(FrameType::MAX_PUSH_ID, (*id).into(), buf),
170            Frame::Grease => {
171                FrameType::grease().encode(buf);
172                buf.write_var(6);
173                buf.put_slice(b"grease");
174            }
175            Frame::WebTransportStream(id) => {
176                FrameType::WEBTRANSPORT_BI_STREAM.encode(buf);
177                id.encode(buf);
178                // rest of the data is sent streaming
179            }
180        }
181    }
182}

377        self.encode_header(buf);

388#[derive(Copy, Clone, PartialEq, Eq, Debug, Hash)]
389pub struct SettingId(pub u64);
390
391impl SettingId {
392    const NONE: SettingId = SettingId(0);
393
394    /// returns a SettingId type with random number of the 0x1f * N + 0x21
395    /// format within the range of the Varint implementation
396    pub fn grease() -> Self {
397        SettingId(fastrand::u64(0..0x210842108421083) * 0x1f + 0x21)
398    }
399
400    fn is_supported(self) -> bool {
401        matches!(
402            self,
403            SettingId::MAX_HEADER_LIST_SIZE
404                | SettingId::QPACK_MAX_TABLE_CAPACITY
405                | SettingId::QPACK_MAX_BLOCKED_STREAMS
406                | SettingId::ENABLE_CONNECT_PROTOCOL
407                | SettingId::ENABLE_WEBTRANSPORT
408                | SettingId::WEBTRANSPORT_MAX_SESSIONS
409                | SettingId::H3_DATAGRAM,
410        )
411    }
412
413    /// Returns if a Settings Identifier is forbidden
414    fn is_forbidden(&self) -> bool {
415        //= https://www.rfc-editor.org/rfc/rfc9114#section-7.2.4.1
416        //# Setting identifiers that were defined in [HTTP/2] where there is no
417        //# corresponding HTTP/3 setting have also been reserved
418        //# (Section 11.2.2).  These reserved settings MUST NOT be sent, and
419        //# their receipt MUST be treated as a connection error of type
420        //# H3_SETTINGS_ERROR.
421        matches!(
422            self,
423            SettingId(0x00) | SettingId(0x02) | SettingId(0x03) | SettingId(0x04) | SettingId(0x05)
424        )
425    }
426
427    fn decode<B: Buf>(buf: &mut B) -> Result<Self, UnexpectedEnd> {
428        Ok(SettingId(buf.get_var()?))
429    }
430
431    fn encode<B: BufMut>(&self, buf: &mut B) {
432        buf.write_var(self.0);
433    }
434}

462#[derive(Debug, PartialEq)]
463pub struct Settings {
464    entries: [(SettingId, u64); SETTINGS_LEN],
465    len: usize,
466}
467
468impl Default for Settings {
469    fn default() -> Self {
470        Self {
471            entries: [(SettingId::NONE, 0); SETTINGS_LEN],
472            len: 0,
473        }
474    }
475}
476
477impl FrameHeader for Settings {
478    const TYPE: FrameType = FrameType::SETTINGS;
479    fn len(&self) -> usize {
480        self.entries[..self.len].iter().fold(0, |len, (id, val)| {
481            len + VarInt::from_u64(id.0).unwrap().size() + VarInt::from_u64(*val).unwrap().size()
482        })
483    }
484}
485
486impl Settings {
487    pub const MAX_ENCODED_SIZE: usize = SETTINGS_LEN * 2 * VarInt::MAX_SIZE;
488
489    pub fn insert(&mut self, id: SettingId, value: u64) -> Result<(), SettingsError> {
490        if self.len >= self.entries.len() {
491            return Err(SettingsError::Exceeded);
492        }
493
494        //= https://www.rfc-editor.org/rfc/rfc9114#section-7.2.4
495        //# The same setting identifier MUST NOT occur more than once in the
496        //# SETTINGS frame.
497        if self.entries[..self.len].iter().any(|(i, _)| *i == id) {
498            return Err(SettingsError::Repeated(id));
499        }
500
501        self.entries[self.len] = (id, value);
502        self.len += 1;
503        Ok(())
504    }
505
506    pub fn get(&self, id: SettingId) -> Option<u64> {
507        for (entry_id, value) in self.entries.iter() {
508            if id == *entry_id {
509                return Some(*value);
510            }
511        }
512        None
513    }
514
515    pub(crate) fn encode<T: BufMut>(&self, buf: &mut T) {
516        self.encode_header(buf);
517        for (id, val) in self.entries[..self.len].iter() {
518            id.encode(buf);
519            buf.write_var(*val);
520        }
521    }
522
523    pub(super) fn decode<T: Buf>(buf: &mut T) -> Result<Settings, SettingsError> {
524        let mut settings = Settings::default();
525        while buf.has_remaining() {
526            if buf.remaining() < 2 {
527                // remains less than 2 * minimum-size varint
528                return Err(SettingsError::Malformed);
529            }
530
531            let identifier = SettingId::decode(buf).map_err(|_| SettingsError::Malformed)?;
532            let value = buf.get_var().map_err(|_| SettingsError::Malformed)?;
533
534            if identifier.is_forbidden() {
535                //= https://www.rfc-editor.org/rfc/rfc9114#section-7.2.4.1
536                //# Setting identifiers that were defined in [HTTP/2] where there is no
537                //# corresponding HTTP/3 setting have also been reserved
538                //# (Section 11.2.2).  These reserved settings MUST NOT be sent, and
539                //# their receipt MUST be treated as a connection error of type
540                //# H3_SETTINGS_ERROR.
541                return Err(SettingsError::InvalidSettingId(identifier.0));
542            }
543
544            if identifier.is_supported() {
545                //= https://www.rfc-editor.org/rfc/rfc9114#section-7.2.4.1
546                //# Setting identifiers that were defined in [HTTP/2] where there is no
547                //# corresponding HTTP/3 setting have also been reserved
548                //# (Section 11.2.2).  These reserved settings MUST NOT be sent, and
549                //# their receipt MUST be treated as a connection error of type
550                //# H3_SETTINGS_ERROR.
551                settings.insert(identifier, value)?;
552            } else {
553                //= https://www.rfc-editor.org/rfc/rfc9114#section-7.2.4.1
554                //# Endpoints MUST NOT consider such settings to have
555                //# any meaning upon receipt.
556                #[cfg(feature = "tracing")]
557                tracing::debug!("Unsupported setting: {:#x?}", identifier);
558            }
559        }
560        Ok(settings)
561    }
562}

587            }

650    fn codec_frame_check(mut frame: Frame<Bytes>, wire: &[u8], check_frame: Frame<Bytes>) {
651        let mut buf = Vec::new();
652        frame.encode_with_payload(&mut buf);
653        assert_eq!(&buf, &wire);
654
655        let mut read = Cursor::new(&buf);
656        let decoded = Frame::decode(&mut read).unwrap();
657        assert_eq!(check_frame, decoded);
658    }

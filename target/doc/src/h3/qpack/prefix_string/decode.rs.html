21impl HuffmanDecoder {
22    fn check_eof(&self, bit_pos: &mut BitWindow, input: &[u8]) -> Result<Option<u32>, Error> {
23        use std::cmp::Ordering;
24        match ((bit_pos.byte + 1) as usize).cmp(&input.len()) {
25            // Position is out-of-range
26            Ordering::Greater => {
27                return Ok(None);
28            }
29            // Position is on the last byte
30            Ordering::Equal => {
31                let side = bit_pos.opposite_bit_window();
32
33                let rest = match read_bits(input, side.byte, side.bit, side.count) {
34                    Ok(x) => x,
35                    Err(()) => {
36                        return Err(Error::MissingBits(side));
37                    }
38                };
39
40                let eof_filler = ((2u16 << (side.count - 1)) - 1) as u8;
41                if rest & eof_filler == eof_filler {
42                    return Ok(None);
43                }
44            }
45            Ordering::Less => {}
46        }
47        Err(Error::MissingBits(bit_pos.clone()))
48    }
49
50    fn fetch_value(&self, bit_pos: &mut BitWindow, input: &[u8]) -> Result<Option<u32>, Error> {
51        match read_bits(input, bit_pos.byte, bit_pos.bit, bit_pos.count) {
52            Ok(value) => Ok(Some(value as u32)),
53            Err(()) => self.check_eof(bit_pos, input),
54        }
55    }
56
57    fn decode_next(&self, bit_pos: &mut BitWindow, input: &[u8]) -> Result<Option<u8>, Error> {
58        bit_pos.forwards(self.lookup);
59
60        let value = match self.fetch_value(bit_pos, input) {
61            Ok(Some(value)) => value as usize,
62            Ok(None) => return Ok(None),
63            Err(err) => return Err(err),
64        };
65
66        let at_value = match (self.table).get(value) {
67            Some(x) => x,
68            None => return Err(Error::Unhandled(bit_pos.clone(), value)),
69        };
70
71        match at_value {
72            DecodeValue::Sym(x) => Ok(Some(*x)),
73            DecodeValue::Partial(d) => d.decode_next(bit_pos, input),
74        }
75    }
76}

78/// Read `len` bits from the `src` slice at the specified position
79///
80/// Never read more than 8 bits at a time. `bit_offset` may be larger than 8.
81fn read_bits(src: &[u8], mut byte_offset: u32, mut bit_offset: u32, len: u32) -> Result<u8, ()> {
82    if len == 0 || len > 8 || src.len() as u32 * 8 < (byte_offset * 8) + bit_offset + len {
83        return Err(());
84    }
85
86    // Deal with `bit_offset` > 8
87    byte_offset += bit_offset / 8;
88    bit_offset -= (bit_offset / 8) * 8;
89
90    Ok(if bit_offset + len <= 8 {
91        // Read all the bits from a single byte
92        (src[byte_offset as usize] << bit_offset) >> (8 - len)
93    } else {
94        // The range of bits spans over 2 bytes
95        let mut result = (src[byte_offset as usize] as u16) << 8;
96        result |= src[byte_offset as usize + 1] as u16;
97        ((result << bit_offset) >> (16 - len)) as u8
98    })
99}

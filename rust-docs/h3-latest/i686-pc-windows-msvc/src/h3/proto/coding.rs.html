10// Trait for encoding / decoding helpers on basic types, such as `u16`, for
11// example: `buf.decode::<u16>()?`.
12// This enables to return `UnexpectedEnd` instead of panicking as the `Buf`
13// impls do when there is not enough bytes.
14
15pub trait Encode {
16    fn encode<B: BufMut>(&self, buf: &mut B);
17}
18
19pub trait Decode: Sized {
20    fn decode<B: Buf>(buf: &mut B) -> Result<Self>;
21}
22
23impl Encode for u8 {
24    fn encode<B: BufMut>(&self, buf: &mut B) {
25        buf.put_u8(*self);
26    }
27}
28
29impl Decode for u8 {
30    fn decode<B: Buf>(buf: &mut B) -> Result<u8> {
31        if buf.remaining() < 1 {
32            return Err(UnexpectedEnd(1));
33        }
34        Ok(buf.get_u8())
35    }
36}
37
38pub trait BufExt {
39    fn get<T: Decode>(&mut self) -> Result<T>;
40    fn get_var(&mut self) -> Result<u64>;
41}
42
43impl<T: Buf> BufExt for T {
44    fn get<U: Decode>(&mut self) -> Result<U> {
45        U::decode(self)
46    }
47
48    fn get_var(&mut self) -> Result<u64> {
49        Ok(VarInt::decode(self)?.into_inner())
50    }
51}
52
53pub trait BufMutExt {
54    fn write<T: Encode>(&mut self, x: T);
55    fn write_var(&mut self, x: u64);
56}
57
58impl<T: BufMut> BufMutExt for T {
59    fn write<U: Encode>(&mut self, x: U) {
60        x.encode(self);
61    }
62
63    fn write_var(&mut self, x: u64) {
64        VarInt::from_u64(x).unwrap().encode(self);
65    }
